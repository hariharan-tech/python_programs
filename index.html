def sum(a,b):
    Sum=[]
    for i in range(len(a)):
        c=[]
        for j in range(len(a[0])):
            c.append(a[i][j]+b[i][j])
        Sum.append(c)
    return Sum

Binary Search 
def binarySearch(arr, l, r, x):
  if r >= l:
      mid = l + (r - l) // 2
      if arr[mid] == x:
          return mid
      elif arr[mid] > x:
          return binarySearch(arr, l, mid - 1, x)
      else:
          return binarySearch(arr, mid + 1, r, x)
  else:
    return -1
